@app.route('/export') def export_data(): depot = request.args.get("depot", "all") filter_type = request.args.get("filter", "all") conn = sqlite3.connect(DB_PATH) c = conn.cursor() query = "SELECT asset_id, name, depot, verified, last_verified, image_path FROM assets WHERE 1=1" params = [] # FILTRO DEPOT if depot != "all": query += " AND depot = ?" params.append(depot) # FILTRO STATUS if filter_type == "verified": query += " AND verified = 1" elif filter_type == "unverified": query += " AND verified = 0" c.execute(query, params) rows = c.fetchall() conn.close() # Criar CSV em memória output = io.StringIO() writer = csv.writer(output) writer.writerow(["Asset ID", "Name", "Depot", "Verified", "Last Verified", "Image Path"]) for row in rows: writer.writerow(row) output.seek(0) return Response( output.getvalue(), mimetype="text/csv", headers={ "Content-Disposition": f"attachment; filename=assets_export_{depot}_{filter_type}.csv" } )
